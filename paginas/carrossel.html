Código: Controle via JavaScript
<!-- 1. O HTML (sem data-bs-ride) -->
<div id="meuCarrossel" class="carousel slide">
  ... (slides aqui) ...
</div>

<!-- Botões de controle externos -->
<button id="btn-prev">Anterior</button>
<button id="btn-next">Próximo</button>
<button id="btn-slide-2">Ir para Slide 2</button>
<button id="btn-pause">Pausar</button>
// 2. O JavaScript (no final da sua página)

// Pega a instância do carrossel
const meuCarrosselEl = document.getElementById('meuCarrossel');
const meuCarrossel = new bootstrap.Carousel(meuCarrosselEl);

// Adiciona eventos aos botões
document.getElementById('btn-prev').addEventListener('click', () => {
  meuCarrossel.prev(); // Move para o anterior
});

document.getElementById('btn-next').addEventListener('click', () => {
  meuCarrossel.next(); // Move para o próximo
});

document.getElementById('btn-pause').addEventListener('click', () => {
  meuCarrossel.pause(); // Pausa o autoplay
});

document.getElementById('btn-slide-2').addEventListener('click', () => {
  meuCarrossel.to(1); // Vai para o slide 2 (índice 1)
});
