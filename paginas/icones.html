Código: Instalação e Uso (Bootstrap Icons)
<!-- 1. Adicione o CDN no <head> da sua página -->
<link href="https://cdn.jsdelivr.net/npm/bootstrap-icons@1.11.3/font/bootstrap-icons.min.css" rel="stylesheet">

<!-- 2. Use a tag <i> com as classes 'bi' e 'bi-NOME-DO-ICONE' -->
<i class="bi bi-bootstrap-fill"></i>
<i class="bi bi-arrow-right-circle"></i>
<i class="bi bi-person-check-fill"></i>
Código: Font Awesome (Alternativa)
<!-- 1. Adicione o CDN do Font Awesome no <head> -->
<link href="https://cdnjs.cloudflare.com/ajax/libs/font-awesome/6.5.1/css/all.min.css" rel="stylesheet">

<!-- 2. Use a tag <i> com as classes de prefixo (ex: fas, fab) -->

<!-- fas = Font Awesome Solid -->
<i class="fas fa-rocket"></i>
<i class="fas fa-heart"></i>

<!-- fab = Font Awesome Brands -->
<i class="fab fa-github"></i>
<i class="fab fa-apple"></i>
Código: Tamanho e Cor
<!-- Para TAMANHO, use as classes de Fonte (fs-*) -->
<i class="bi bi-star-fill fs-1"></i> <!-- Grande -->
<i class="bi bi-star-fill fs-3"></i> <!-- Médio -->
<i class="bi bi-star-fill fs-5"></i> <!-- Pequeno -->

<!-- Ou use style="font-size: ..." -->
<i class="bi bi-star-fill" style="font-size: 50px;"></i>

<!-- Para COR, use as classes de Texto (text-*) -->
<i class="bi bi-heart-fill fs-2 text-primary"></i>
<i class="bi bi-heart-fill fs-2 text-success"></i>
<i class="bi bi-heart-fill fs-2 text-danger"></i>
<i class="bi bi-heart-fill fs-2 text-warning"></i>
Código: Ícones em Botões
<!-- Adicione um <i> dentro do <button> -->
<!-- Use 'me-2' (margin-end) para dar espaço -->
<button type="button" class="btn btn-primary">
  <i class="bi bi-save me-2"></i>Salvar
</button>

<button type="button" class="btn btn-danger">
  <i class="bi bi-trash me-2"></i>Excluir
</button>

<button type="button" class="btn btn-info">
  <i class="bi bi-info-circle me-2"></i>Info
</button>

<!-- Apenas ícone (requer 'aria-label' para acessibilidade) -->
<button type="button" class="btn btn-outline-secondary" aria-label="Procurar">
  <i class="bi bi-search"></i>
</button>
Código: Ícones em Input Groups
<!-- Coloque o ícone dentro de um .input-group-text -->

<!-- Ícone antes -->
<div class="input-group mb-3">
  <span class="input-group-text">
    <i class="bi bi-person-fill"></i>
  </span>
  <input type="text" class="form-control" placeholder="Nome de usuário">
</div>

<!-- Ícone depois -->
<div class="input-group mb-3">
  <input type="text" class="form-control" placeholder="E-mail">
  <span class="input-group-text">
    <i class="bi bi-envelope-fill"></i>
  </span>
</div>
